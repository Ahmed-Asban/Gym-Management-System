using System;
using System.Windows.Forms;

public partial class Form1 : Form
{
    public Form1()
    {
        InitializeComponent();
    }

    private void button1_Click(object sender, EventArgs e)
    {
        // Get the specific date from the date picker
        DateTime startDate = dateTimePicker1.Value;

        // Calculate the payment date one month from the starting date
        DateTime paymentDate = startDate.AddMonths(1);

        // Check if the payment date falls in January
        if (paymentDate.Month != 1)
        {
            // If not, adjust the payment date to the next January
            paymentDate = paymentDate.AddYears(1).SetTime(0, 0, 0, 0);
        }

        // Calculate the number of days remaining until the payment date
        TimeSpan remainingDays = paymentDate - DateTime.Today;

        // Display the payment date and the number of remaining days
        labelPaymentDate.Text = paymentDate.ToString("MM/dd/yyyy");
        labelRemainingDays.Text = remainingDays.Days.ToString();
    }
